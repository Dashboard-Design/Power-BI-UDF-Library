	/// Returns a country flag image URL from a country name.
	///
	/// This function converts a country name (for example: "France",
	/// "Canada", "United States", "Germany") into its ISO 3166-1 alpha-2
	/// country code and returns a URL that can be used directly in:
	/// Parameters:
	/// Country_dimensionColumn : ANYREF EXPR
	///     Country name column or expression, Examples: Customer[Country]
	/// Style : SCALAR NUMERIC VAL
	///     Determines which flag style is returned.
	function CountryFlag  = ```
		( 
			Country_dimensionColumn : ANYREF EXPR,
			Style: SCALAR NUMERIC VAL
		) =>

		VAR _NormalizedInput =
			LOWER( SELECTEDVALUE(Country_dimensionColumn) )

		VAR _SelectedStyle =
			COALESCE( Style, 1 )
